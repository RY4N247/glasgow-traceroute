1//! Implementation of the [ChaCha] family of stream ciphers.
2//!
3//! Cipher functionality is accessed using traits from re-exported [`cipher`] crate.
4//!
5//! ChaCha stream ciphers are lightweight and amenable to fast, constant-time
6//! implementations in software. It improves upon the previous [Salsa] design,
7//! providing increased per-round diffusion with no cost to performance.
8//!
9//! This crate contains the following variants of the ChaCha20 core algorithm:
10//!
11//! - [`ChaCha20`]: standard IETF variant with 96-bit nonce
12//! - [`ChaCha8`] / [`ChaCha12`]: reduced round variants of ChaCha20
13//! - [`XChaCha20`]: 192-bit extended nonce variant
14//! - [`XChaCha8`] / [`XChaCha12`]: reduced round variants of XChaCha20
15//! - [`ChaCha20Legacy`]: "djb" variant with 64-bit nonce.
16//! **WARNING:** This implementation internally uses 32-bit counter,
17//! while the original implementation uses 64-bit counter. In other words,
18//! it does not allow encryption of more than 256 GiB of data.
19//!
20//! # ⚠️ Security Warning: Hazmat!
21//!
22//! This crate does not ensure ciphertexts are authentic, which can lead to
23//! serious vulnerabilities if used incorrectly!
24//!
25//! If in doubt, use the [`chacha20poly1305`] crate instead, which provides
26//! an authenticated mode on top of ChaCha20.
27//!
28//! **USE AT YOUR OWN RISK!**
29//!
30//! # Diagram
31//!
32//! This diagram illustrates the ChaCha quarter round function.
33//! Each round consists of four quarter-rounds:
34//!
35//! <img src="https://raw.githubusercontent.com/RustCrypto/media/8f1a9894/img/stream-ciphers/chacha20.png" width="300px">
36//!
37//! Legend:
38//!
39//! - ⊞ add
40//! - ‹‹‹ rotate
41//! - ⊕ xor
42//!
43//! # Example
44#![cfg_attr(feature = "cipher", doc = " ```")]
45#![cfg_attr(not(feature = "cipher"), doc = " ```ignore")]
46//! use chacha20::ChaCha20;
47//! // Import relevant traits
48//! use chacha20::cipher::{KeyIvInit, StreamCipher, StreamCipherSeek};
49//! use hex_literal::hex;
50//!
51//! let key = [0x42; 32];
52//! let nonce = [0x24; 12];
53//! let plaintext = hex!("00010203 04050607 08090A0B 0C0D0E0F");
54//! let ciphertext = hex!("e405626e 4f1236b3 670ee428 332ea20e");
55//!
56//! // Key and IV must be references to the `Array` type.
57//! // Here we use the `Into` trait to convert arrays into it.
58//! let mut cipher = ChaCha20::new(&key.into(), &nonce.into());
59//!
60//! let mut buffer = plaintext.clone();
61//!
62//! // apply keystream (encrypt)
63//! cipher.apply_keystream(&mut buffer);
64//! assert_eq!(buffer, ciphertext);
65//!
66//! let ciphertext = buffer.clone();
67//!
68//! // ChaCha ciphers support seeking
69//! cipher.seek(0u32);
70//!
71//! // decrypt ciphertext by applying keystream again
72//! cipher.apply_keystream(&mut buffer);
73//! assert_eq!(buffer, plaintext);
74//!
75//! // stream ciphers can be used with streaming messages
76//! cipher.seek(0u32);
77//! for chunk in buffer.chunks_mut(3) {
78//!     cipher.apply_keystream(chunk);
79//! }
80//! assert_eq!(buffer, ciphertext);
81//! ```
82//!
83//! # Configuration Flags
84//!
85//! You can modify crate using the following configuration flags:
86//!
87//! - `chacha20_force_avx2`: force AVX2 backend on x86/x86_64 targets.
88//!   Requires enabled AVX2 target feature. Ignored on non-x86(-64) targets.
89//! - `chacha20_force_soft`: force software backend.
90//! - `chacha20_force_sse2`: force SSE2 backend on x86/x86_64 targets.
91//!   Requires enabled SSE2 target feature. Ignored on non-x86(-64) targets.
92//!
93//! The flags can be enabled using `RUSTFLAGS` environmental variable
94//! (e.g. `RUSTFLAGS="--cfg chacha20_force_avx2"`) or by modifying `.cargo/config`.
95//!
96//! You SHOULD NOT enable several `force` flags simultaneously.
97//!
98//! [ChaCha]: https://tools.ietf.org/html/rfc8439
99//! [Salsa]: https://en.wikipedia.org/wiki/Salsa20
100//! [`chacha20poly1305`]: https://docs.rs/chacha20poly1305

183cfg_if! {
184    if #[cfg(chacha20_force_soft)] {
185        type Tokens = ();
186    } else if #[cfg(any(target_arch = "x86", target_arch = "x86_64"))] {
187        cfg_if! {
188            if #[cfg(chacha20_force_avx2)] {
189                #[cfg(not(target_feature = "avx2"))]
190                compile_error!("You must enable `avx2` target feature with \
191                    `chacha20_force_avx2` configuration option");
192                type Tokens = ();
193            } else if #[cfg(chacha20_force_sse2)] {
194                #[cfg(not(target_feature = "sse2"))]
195                compile_error!("You must enable `sse2` target feature with \
196                    `chacha20_force_sse2` configuration option");
197                type Tokens = ();
198            } else {
199                cpufeatures::new!(avx2_cpuid, "avx2");
200                cpufeatures::new!(sse2_cpuid, "sse2");
201                type Tokens = (avx2_cpuid::InitToken, sse2_cpuid::InitToken);
202            }
203        }
204    } else {
205        type Tokens = ();
206    }
207}

209/// The ChaCha core function.
210pub struct ChaChaCore<R: Rounds, V: Variant> {
211    /// Internal state of the core function
212    #[cfg(any(feature = "cipher", feature = "rng"))]
213    state: [u32; STATE_WORDS],
214    /// CPU target feature tokens
215    #[allow(dead_code)]
216    tokens: Tokens,
217    /// Number of rounds to perform and the cipher variant
218    _pd: PhantomData<(R, V)>,
219}
220
221impl<R: Rounds, V: Variant> ChaChaCore<R, V> {
222    /// Constructs a ChaChaCore with the specified key, iv, and amount of rounds.
223    /// You must ensure that the iv is of the correct size when using this method
224    /// directly.
225    #[cfg(any(feature = "cipher", feature = "rng"))]
226    fn new(key: &[u8; 32], iv: &[u8]) -> Self {
227        let mut state = [0u32; STATE_WORDS];
228
229        let ctr_size = size_of::<V::Counter>() / size_of::<u32>();
230        let (const_dst, state_rem) = state.split_at_mut(4);
231        let (key_dst, state_rem) = state_rem.split_at_mut(8);
232        let (_ctr_dst, iv_dst) = state_rem.split_at_mut(ctr_size);
233
234        const_dst.copy_from_slice(&CONSTANTS);
235
236        for (src, dst) in key.chunks_exact(4).zip(key_dst) {
237            *dst = u32::from_le_bytes(src.try_into().unwrap());
238        }
239
240        assert_eq!(size_of_val(iv_dst), size_of_val(iv));
241        for (src, dst) in iv.chunks_exact(4).zip(iv_dst) {
242            *dst = u32::from_le_bytes(src.try_into().unwrap());
243        }
244
245        cfg_if! {
246            if #[cfg(chacha20_force_soft)] {
247                let tokens = ();
248            } else if #[cfg(any(target_arch = "x86", target_arch = "x86_64"))] {
249                cfg_if! {
250                    if #[cfg(chacha20_force_avx2)] {
251                        let tokens = ();
252                    } else if #[cfg(chacha20_force_sse2)] {
253                        let tokens = ();
254                    } else {
255                        let tokens = (avx2_cpuid::init(), sse2_cpuid::init());
256                    }
257                }
258            } else {
259                let tokens = ();
260            }
261        }
262        Self {
263            state,
264            tokens,
265            _pd: PhantomData,
266        }
267    }
268}

285#[cfg(feature = "cipher")]
286impl<R: Rounds, V: Variant> StreamCipherCore for ChaChaCore<R, V> {
287    #[inline(always)]
288    fn remaining_blocks(&self) -> Option<usize> {
289        V::remaining_blocks(self.get_block_pos())
290    }
291
292    fn process_with_backend(
293        &mut self,
294        f: impl cipher::StreamCipherClosure<BlockSize = Self::BlockSize>,
295    ) {
296        cfg_if! {
297            if #[cfg(chacha20_force_soft)] {
298                f.call(&mut backends::soft::Backend(self));
299            } else if #[cfg(any(target_arch = "x86", target_arch = "x86_64"))] {
300                cfg_if! {
301                    if #[cfg(chacha20_force_avx2)] {
302                        unsafe {
303                            backends::avx2::inner::<R, _, V>(&mut self.state, f);
304                        }
305                    } else if #[cfg(chacha20_force_sse2)] {
306                        unsafe {
307                            backends::sse2::inner::<R, _, V>(&mut self.state, f);
308                        }
309                    } else {
310                        let (avx2_token, sse2_token) = self.tokens;
311                        if avx2_token.get() {
312                            unsafe {
313                                backends::avx2::inner::<R, _, V>(&mut self.state, f);
314                            }
315                        } else if sse2_token.get() {
316                            unsafe {
317                                backends::sse2::inner::<R, _, V>(&mut self.state, f);
318                            }
319                        } else {
320                            f.call(&mut backends::soft::Backend(self));
321                        }
322                    }
323                }
324            } else if #[cfg(all(target_arch = "aarch64", target_feature = "neon"))] {
325                unsafe {
326                    backends::neon::inner::<R, _, V>(&mut self.state, f);
327                }
328            } else {
329                f.call(&mut backends::soft::Backend(self));
330            }
331        }
332    }
333}

350/// The ChaCha20 quarter round function
351///
352/// We located this function in the root of the crate as we want it to be available
353/// for the soft backend and for xchacha.
354#[allow(dead_code)]
355pub(crate) fn quarter_round(
356    a: usize,
357    b: usize,
358    c: usize,
359    d: usize,
360    state: &mut [u32; STATE_WORDS],
361) {
362    state[a] = state[a].wrapping_add(state[b]);
363    state[d] ^= state[a];
364    state[d] = state[d].rotate_left(16);
365
366    state[c] = state[c].wrapping_add(state[d]);
367    state[b] ^= state[c];
368    state[b] = state[b].rotate_left(12);
369
370    state[a] = state[a].wrapping_add(state[b]);
371    state[d] ^= state[a];
372    state[d] = state[d].rotate_left(8);
373
374    state[c] = state[c].wrapping_add(state[d]);
375    state[b] ^= state[c];
376    state[b] = state[b].rotate_left(7);
377}
