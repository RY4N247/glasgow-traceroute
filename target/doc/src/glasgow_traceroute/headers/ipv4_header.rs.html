26impl Ipv4Header {
27    pub fn build_packet(&mut self, payload: &[u8], mode: Option<ByteOrderMode>) -> Vec<u8> {
28        self.set_total_length_from_payload(payload.len());
29        self.set_new_id();
30
31        let mut packet = match mode.unwrap_or(ByteOrderMode::Auto) {
32            ByteOrderMode::Network => self.to_byte_array_network(),
33            ByteOrderMode::Auto => self.to_byte_array(), // platform-aware
34        };
35
36        packet.extend_from_slice(payload);
37        packet
38    }
39
40    pub fn to_byte_array(&self) -> Vec<u8> {
41        let mut buf: Vec<u8> = Vec::with_capacity(20);
42
43        // Version 4 bits + IHL 4 bits
44        buf.push((self.version << 4) | (self.internet_header_length & 0b00001111));
45
46        //type of service 6 bits + ecn 2 bits
47        buf.push((self.type_of_service << 2) | (self.explicit_congestion_notification & 0b00000011));
48
49        // ip_len is in host byte order as per man ip 4
50        #[cfg(target_os = "macos")]
51        buf.extend_from_slice(&self.total_length.to_ne_bytes());
52
53        #[cfg(target_os = "linux")]
54        buf.extend_from_slice(&self.total_length.to_be_bytes());
55
56        buf.extend_from_slice(&self.identification.to_be_bytes());
57
58        let flags_u16 = (self.flags.to_u8() as u16) << 13;
59        let offset_u16 = self.fragment_offset & 0b0001_1111_1111_1111;
60        let combined = flags_u16 | offset_u16;
61
62        //ip_len is in host byte order as per man ip 4
63        #[cfg(target_os = "macos")]
64        buf.extend_from_slice(&combined.to_ne_bytes());
65
66        #[cfg(target_os = "linux")]
67        buf.extend_from_slice(&combined.to_be_bytes());
68
69        buf.push(self.time_to_live);
70
71        buf.push(self.protocol.to_u8());
72
73        buf.extend_from_slice(&[0,0]);
74
75        buf.extend_from_slice(&self.source_address.octets());
76
77        buf.extend_from_slice(&self.destination_address.octets());
78
79        //compute checksum and
80        let checksum = checksum(&buf[..20]);
81
82        // Insert checksum into bytes 10 and 11
83        buf[10] = (checksum >> 8) as u8;
84        buf[11] = (checksum & 0xFF) as u8;
85
86        buf
87    }
88    pub fn to_byte_array_network(&self) -> Vec<u8> {
89        let mut buf: Vec<u8> = Vec::with_capacity(20);
90
91        buf.push((self.version << 4) | (self.internet_header_length & 0b00001111));
92
93        buf.push((self.type_of_service << 2) | (self.explicit_congestion_notification & 0b00000011));
94
95        buf.extend_from_slice(&self.total_length.to_be_bytes());
96
97        buf.extend_from_slice(&self.identification.to_be_bytes());
98
99        let flags_u16 = (self.flags.to_u8() as u16) << 13;
100        let offset_u16 = self.fragment_offset & 0b0001_1111_1111_1111;
101        let combined = flags_u16 | offset_u16;
102
103        buf.extend_from_slice(&combined.to_be_bytes());
104
105        buf.push(self.time_to_live);
106
107        buf.push(self.protocol.to_u8());
108
109        buf.extend_from_slice(&[0,0]);
110
111        buf.extend_from_slice(&self.source_address.octets());
112
113        buf.extend_from_slice(&self.destination_address.octets());
114
115        //compute checksum and
116        let checksum = checksum(&buf[..20]);
117
118        // Insert checksum into bytes 10 and 11
119        buf[10] = (checksum >> 8) as u8;
120        buf[11] = (checksum & 0xFF) as u8;
121
122        buf
123    }
124    pub fn set_total_length_from_payload(&mut self, payload_length: usize) {
125        self.total_length = (20 + payload_length) as u16;
126    }
127    pub fn set_new_id(&mut self) {
128        self.identification = rand::random::<u16>();
129    }
