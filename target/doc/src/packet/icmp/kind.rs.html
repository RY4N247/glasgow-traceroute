15/// ICMP packet types.
16#[derive(Eq, PartialEq, Copy, Clone, Debug)]
17pub enum Kind {
18	///
19	EchoReply,
20
21	///
22	DestinationUnreachable,
23
24	///
25	SourceQuench,
26
27	///
28	RedirectMessage,
29
30	///
31	EchoRequest,
32
33	///
34	RouterAdvertisement,
35
36	///
37	RouterSolicitation,
38
39	///
40	TimeExceeded,
41
42	///
43	ParameterProblem,
44
45	///
46	TimestampRequest,
47
48	///
49	TimestampReply,
50
51	///
52	InformationRequest,
53
54	///
55	InformationReply,
56
57	///
58	AddressMaskRequest,
59
60	///
61	AddressMaskReply,
62
63	///
64	TraceRoute,
65
66	///
67	Unknown(u8),
68}
69
70impl From<u8> for Kind {
71	fn from(value: u8) -> Kind {
72		use self::Kind::*;
73
74		match value {
75			0  => EchoReply,
76			3  => DestinationUnreachable,
77			4  => SourceQuench,
78			5  => RedirectMessage,
79			8  => EchoRequest,
80			9  => RouterAdvertisement,
81			10 => RouterSolicitation,
82			11 => TimeExceeded,
83			12 => ParameterProblem,
84			13 => TimestampRequest,
85			14 => TimestampReply,
86			15 => InformationRequest,
87			16 => InformationReply,
88			17 => AddressMaskRequest,
89			18 => AddressMaskReply,
90			30 => TraceRoute,
91			v  => Unknown(v),
92		}
93	}
94}
95
96impl Into<u8> for Kind {
97	fn into(self) -> u8 {
98		use self::Kind::*;
99
100		match self {
101			EchoReply              => 0,
102			DestinationUnreachable => 3,
103			SourceQuench           => 4,
104			RedirectMessage        => 5,
105			EchoRequest            => 8,
106			RouterAdvertisement    => 9,
107			RouterSolicitation     => 10,
108			TimeExceeded           => 11,
109			ParameterProblem       => 12,
110			TimestampRequest       => 13,
111			TimestampReply         => 14,
112			InformationRequest     => 15,
113			InformationReply       => 16,
114			AddressMaskRequest     => 17,
115			AddressMaskReply       => 18,
116			TraceRoute             => 30,
117			Unknown(v)             => v,
118		}
119	}
120}
