91impl From<u16> for Protocol {
92	fn from(value: u16) -> Protocol {
93		use self::Protocol::*;
94
95		match value {
96			0x0800 => Ipv4,
97			0x0806 => Arp,
98			0x0842 => WakeOnLan,
99			0x22f3 => Trill,
100			0x6003 => DecNet,
101			0x8035 => Rarp,
102			0x809b => AppleTalk,
103			0x80f3 => Aarp,
104			0x8137 => Ipx,
105			0x8204 => Qnx,
106			0x86dd => Ipv6,
107			0x8808 => FlowControl,
108			0x8819 => CobraNet,
109			0x8847 => Mpls,
110			0x8848 => MplsMulticast,
111			0x8863 => PppoeDiscovery,
112			0x8864 => PppoeSession,
113			0x8100 => Vlan,
114			0x88a8 => PBridge,
115			0x88cc => Lldp,
116			0x88f7 => Ptp,
117			0x8902 => Cfm,
118			0x9100 => QinQ,
119			n      => Unknown(n),
120		}
121	}
122}
