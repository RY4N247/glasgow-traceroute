15use std::ops::{Deref, DerefMut};
16
17use crate::error::*;
18
19/// A growable buffer.
20#[derive(Clone, Eq, PartialEq, Default, Debug)]
21pub struct Buffer {
22	inner: Vec<u8>,
23
24	offset: usize,
25	length: usize,
26}
27
28impl Buffer {
29	/// Create a new growable buffer.
30	pub fn new() -> Self {
31		Default::default()
32	}
33}
34
35impl super::Buffer for Buffer {
36	type Inner = Vec<u8>;
37
38	fn into_inner(self) -> Self::Inner {
39		self.inner
40	}
41
42	fn next(&mut self, size: usize) -> Result<()> {
43		self.offset += self.length;
44		self.length  = size;
45
46		let current = self.inner.len();
47		self.inner.resize(current + size, 0);
48
49		Ok(())
50	}
51
52	fn more(&mut self, size: usize) -> Result<()> {
53		let current = self.inner.len();
54		self.inner.resize(current + size, 0);
55		self.length += size;
56
57		Ok(())
58	}
59
60	fn clear(&mut self) {
61		self.inner.clear();
62		self.offset = 0;
63		self.length = 0;
64	}
65
66	fn used(&self) -> usize {
67		self.inner.len()
68	}
69
70	fn offset(&self) -> usize {
71		self.offset
72	}
73
74	fn length(&self) -> usize {
75		self.length
76	}
77
78	fn data(&self) -> &[u8] {
79		&self.inner[self.offset .. self.offset + self.length]
80	}
81
82	fn data_mut(&mut self) -> &mut [u8] {
83		&mut self.inner[self.offset .. self.offset + self.length]
84	}
85}
86
87impl Into<Vec<u8>> for Buffer {
88	fn into(self) -> Vec<u8> {
89		self.inner
90	}
91}
92
93impl AsRef<[u8]> for Buffer {
94	fn as_ref(&self) -> &[u8] {
95		use super::Buffer;
96		self.data()
97	}
98}
99
100impl AsMut<[u8]> for Buffer {
101	fn as_mut(&mut self) -> &mut [u8] {
102		use super::Buffer;
103		self.data_mut()
104	}
105}
106
107impl Deref for Buffer {
108	type Target = [u8];
109
110	fn deref(&self) -> &Self::Target {
111		use super::Buffer;
112		self.data()
113	}
114}
115
116impl DerefMut for Buffer {
117	fn deref_mut(&mut self) -> &mut Self::Target {
118		use super::Buffer;
119		self.data_mut()
120	}
121}
