15use bitflags::bitflags;
16
17bitflags! {
18	/// IPv4 packet flags.
19	pub struct Flags: u16 {
20		/// Do not fragment packets.
21		const DONT_FRAGMENT = 0b010;
22
23		/// More fragments are waiting.
24		const MORE_FRAGMENTS = 0b100;
25	}
26}
27
28pub const DONT_FRAGMENT: Flags  = Flags::DONT_FRAGMENT;
29pub const MORE_FRAGMENTS: Flags = Flags::MORE_FRAGMENTS;
