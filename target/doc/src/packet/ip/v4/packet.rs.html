27/// IPv4 packet parser.
28#[derive(Copy, Clone)]
29pub struct Packet<B> {
30	buffer: B,
31}
32
33sized!(Packet,
34	header {
35		min:  20,
36		max:  60,
37		size: p => p.header() as usize * 4,
38	}
39
40	payload {
41		min:  0,
42		max:  u16::max_value() as usize - 60,
43		size: p => (p.length() as usize).saturating_sub(p.header() as usize * 4),
44	});
45
46impl<B: AsRef<[u8]>> fmt::Debug for Packet<B> {
47	fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
48		f.debug_struct(if self.is_valid() { "ip::v4::Packet" } else { "ip::v4::Packet!" })
49			.field("version", &self.version())
50			.field("header", &self.header())
51			.field("dscp", &self.dscp())
52			.field("ecn", &self.ecn())
53			.field("length", &self.length())
54			.field("id", &self.id())
55			.field("flags", &self.flags())
56			.field("offset", &self.offset())
57			.field("ttl", &self.ttl())
58			.field("protocol", &self.protocol())
59			.field("checksum", &self.checksum())
60			.field("source", &self.source())
61			.field("destination", &self.destination())
62			.field("payload", &self.payload())
63			.finish()
64	}
65}

67impl<B: AsRef<[u8]>> Packet<B> {
68	/// Create an IPv4 packet without checking the buffer.
69	pub fn unchecked(buffer: B) -> Packet<B> {
70		Packet { buffer }
71	}
72
73	/// Parse an IPv4 packet without checking the payload.
74	pub fn no_payload(buffer: B) -> Result<Packet<B>> {
75		use crate::size::header::Min;
76
77		let packet = Packet::unchecked(buffer);
78
79		if packet.buffer.as_ref().len() < Self::min() {
80			Err(Error::SmallBuffer)?
81		}
82
83		if packet.buffer.as_ref()[0] >> 4 != 4 {
84			Err(Error::InvalidPacket)?
85		}
86
87		if packet.buffer.as_ref().len() < packet.header() as usize * 4 {
88			Err(Error::SmallBuffer)?
89		}
90
91		Ok(packet)
92	}
93
94	/// Parse an IPv4 packet, checking the buffer contents are correct.
95	pub fn new(buffer: B) -> Result<Packet<B>> {
96		let packet = Packet::no_payload(buffer)?;
97
98		if packet.buffer.as_ref().len() < packet.length() as usize {
99			Err(Error::SmallBuffer)?
100		}
101
102		Ok(packet)
103	}
104}

147impl<B: AsRef<[u8]>> P for Packet<B> {
148	fn split(&self) -> (&[u8], &[u8]) {
149		use crate::size::payload::Size;
150
151		let header  = self.header() as usize * 4;
152		let payload = self.size();
153
154		let buffer = self.buffer.as_ref();
155		let buffer = if buffer.len() < header + payload {
156			buffer
157		}
158		else {
159			&buffer[.. header + payload]
160		};
161
162		buffer.split_at(header)
163	}
164}

166impl<B: AsRef<[u8]> + AsMut<[u8]>> PM for Packet<B> {
167	fn split_mut(&mut self) -> (&mut [u8], &mut [u8]) {
168		use crate::size::payload::Size;
169
170		let header  = self.header() as usize * 4;
171		let payload = self.size();
172
173		let buffer = self.buffer.as_mut();
174		let buffer = if buffer.len() < header + payload {
175			buffer
176		}
177		else {
178			&mut buffer[.. header + payload]
179		};
180
181		buffer.split_at_mut(header)
182	}
183}

185impl<B: AsRef<[u8]>> Packet<B> {
186	/// IP protocol version, will always be 4.
187	pub fn version(&self) -> u8 {
188		self.buffer.as_ref()[0] >> 4
189	}
190
191	/// Length of the IPv4 header in 32 bit words.
192	pub fn header(&self) -> u8 {
193		self.buffer.as_ref()[0] & 0b1111
194	}
195
196	/// DSCP value.
197	pub fn dscp(&self) -> u8 {
198		self.buffer.as_ref()[1] >> 2
199	}
200
201	/// ECN value.
202	pub fn ecn(&self) -> u8 {
203		self.buffer.as_ref()[1] & 0b11
204	}
205
206	/// Total length of the packet in octets.
207	pub fn length(&self) -> u16 {
208		(&self.buffer.as_ref()[2 ..]).read_u16::<BigEndian>().unwrap()
209	}
210
211	/// ID of the packet.
212	pub fn id(&self) -> u16 {
213		(&self.buffer.as_ref()[4 ..]).read_u16::<BigEndian>().unwrap()
214	}
215
216	/// Flags of the packet.
217	pub fn flags(&self) -> Flags {
218		Flags::from_bits((&self.buffer.as_ref()[6 ..])
219			.read_u16::<BigEndian>().unwrap() >> 13).unwrap()
220	}
221
222	/// Offset of the packet.
223	pub fn offset(&self) -> u16 {
224		(&self.buffer.as_ref()[6 ..]).read_u16::<BigEndian>().unwrap() & 0x1fff
225	}
226
227	/// Time to Live for the packet.
228	pub fn ttl(&self) -> u8 {
229		self.buffer.as_ref()[8]
230	}
231
232	/// Protocol of the inner packet.
233	pub fn protocol(&self) -> Protocol {
234		self.buffer.as_ref()[9].into()
235	}
236
237	/// Checksum of the packet.
238	pub fn checksum(&self) -> u16 {
239		(&self.buffer.as_ref()[10 ..]).read_u16::<BigEndian>().unwrap()
240	}
241
242	/// Verify the packet is valid by calculating the checksum.
243	pub fn is_valid(&self) -> bool {
244		checksum(P::header(self)) == self.checksum()
245	}
246
247	/// Source IP address.
248	pub fn source(&self) -> Ipv4Addr {
249		Ipv4Addr::new(
250			self.buffer.as_ref()[12],
251			self.buffer.as_ref()[13],
252			self.buffer.as_ref()[14],
253			self.buffer.as_ref()[15])
254	}
255
256	/// Destination IP address.
257	pub fn destination(&self) -> Ipv4Addr {
258		Ipv4Addr::new(
259			self.buffer.as_ref()[16],
260			self.buffer.as_ref()[17],
261			self.buffer.as_ref()[18],
262			self.buffer.as_ref()[19])
263	}
264
265	/// IP options for the packet.
266	pub fn options(&self) -> OptionIter<'_> {
267		OptionIter {
268			buffer: &self.buffer.as_ref()[20 .. (self.header() as usize * 4)],
269		}
270	}
271}

273impl<B: AsRef<[u8]> + AsMut<[u8]>> Packet<B> {
274	/// Differentiated Services Code Point.
275	pub fn set_dscp(&mut self, value: u8) -> Result<&mut Self> {
276		if value > 0b11_1111 {
277			Err(Error::InvalidValue)?
278		}
279
280		let old = self.buffer.as_ref()[1];
281		self.buffer.as_mut()[1] = (old & 0b11) | value << 2;
282
283		Ok(self)
284	}
285
286	/// Explicit Congestion Notification.
287	pub fn set_ecn(&mut self, value: u8) -> Result<&mut Self> {
288		if value > 0b11 {
289			Err(Error::InvalidValue)?
290		}
291
292		let old = self.buffer.as_ref()[1];
293		self.buffer.as_mut()[1] = (old & 0b11_1111) | value;
294
295		Ok(self)
296	}
297
298	/// Packet ID.
299	pub fn set_id(&mut self, value: u16) -> Result<&mut Self> {
300		Cursor::new(&mut self.buffer.as_mut()[4 ..])
301			.write_u16::<BigEndian>(value)?;
302
303		Ok(self)
304	}
305
306	/// Packet flags.
307	pub fn set_flags(&mut self, value: Flags) -> Result<&mut Self> {
308		Cursor::new(&mut self.header_mut()[6 ..])
309			.write_u16::<BigEndian>(value.bits())?;
310
311		Ok(self)
312	}
313
314	/// Packet fragment offset.
315	pub fn set_offset(&mut self, value: u16) -> Result<&mut Self> {
316		Cursor::new(&mut self.header_mut()[6 ..])
317			.write_u16::<BigEndian>(value)?;
318
319		Ok(self)
320	}
321
322	/// Time to Live.
323	pub fn set_ttl(&mut self, value: u8) -> Result<&mut Self> {
324		self.header_mut()[8] = value;
325
326		Ok(self)
327	}
328
329	/// Source address.
330	pub fn set_source(&mut self, value: Ipv4Addr) -> Result<&mut Self> {
331		self.header_mut()[12 .. 16].copy_from_slice(&value.octets());
332
333		Ok(self)
334	}
335
336	/// Destination address.
337	pub fn set_destination(&mut self, value: Ipv4Addr) -> Result<&mut Self> {
338		self.header_mut()[16 .. 20].copy_from_slice(&value.octets());
339
340		Ok(self)
341	}
342
343	/// Inner protocol.
344	pub fn set_protocol(&mut self, value: Protocol) -> Result<&mut Self> {
345		self.header_mut()[9] = value.into();
346
347		Ok(self)
348	}
349
350	/// Create a checksumed setter.
351	pub fn checked(&mut self) -> Checked<'_, B> {
352		Checked {
353			packet: self
354		}
355	}
356
357	/// Set the checksum value.
358	pub fn set_checksum(&mut self, value: u16) -> Result<&mut Self> {
359		Cursor::new(&mut self.header_mut()[10 ..])
360			.write_u16::<BigEndian>(value)?;
361
362		Ok(self)
363	}
364
365	/// Recalculate and set the checksum value.
366	pub fn update_checksum(&mut self) -> Result<&mut Self> {
367		let checksum = checksum(P::header(self));
368		self.set_checksum(checksum)
369	}
370}

443/// Iterator over IP packet options.
444pub struct OptionIter<'a> {
445	buffer: &'a [u8],
446}
447
448impl<'a> Iterator for OptionIter<'a> {
449	type Item = Result<option::Option<&'a [u8]>>;
450
451	fn next(&mut self) -> Option<Self::Item> {
452		use crate::size::Size;
453
454		if self.buffer.is_empty() {
455			return None;
456		}
457
458		match option::Option::new(self.buffer) {
459			Ok(option) => {
460				if option.number() == option::Number::End {
461					return None;
462				}
463
464				self.buffer = &self.buffer[option.size() ..];
465				Some(Ok(option))
466			}
467
468			Err(error) =>
469				Some(Err(error))
470		}
471	}
472}

480	fn short_packet() {
481		assert!(ip::v4::Packet::no_payload(&[64; 10][..]).is_err());

486	#[test]
487	fn values() {
488		let raw = [0x45u8, 0x00, 0x00, 0x34, 0x2d, 0x87, 0x00, 0x00, 0x2c, 0x06, 0x5c, 0x74, 0x42, 0x66, 0x01, 0x6c, 0xc0, 0xa8, 0x00, 0x4f];
489		let ip  = ip::v4::Packet::no_payload(&raw[..]).unwrap();
490
491		assert_eq!(ip.header(), 5);
492		assert_eq!(ip.length(), 52);
493		assert_eq!(ip.id(), 0x2d87);
494		assert!(ip.flags().is_empty());
495		assert_eq!(ip.protocol(), ip::Protocol::Tcp);
496		assert_eq!(ip.checksum(), 0x5c74);
497		assert!(ip.is_valid());
498		assert_eq!(ip.source(), "66.102.1.108".parse::<Ipv4Addr>().unwrap());
499		assert_eq!(ip.destination(), "192.168.0.79".parse::<Ipv4Addr>().unwrap());
